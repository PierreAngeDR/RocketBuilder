<?php

namespace App\DataPersister;

use ApiPlatform\Metadata\Operation;
use ApiPlatform\State\ProcessorInterface;
use Symfony\Bundle\SecurityBundle\Security;
use App\Entity\User;

abstract class RocketDataPersister implements ProcessorInterface
{
    protected string $resourceClass;
    protected array $methods;

    public function __construct(
        private readonly ProcessorInterface $decorated,
        private readonly Security $security
    ) {
        $this->setProcessParameters();
    }

    /**
     * Ã€ dÃ©finir dans chaque sous-classe :
     *  - $this->resourceClass = NomDeLEntitÃ©::class
     *  - $this->methods = [ 'getXxx' => 'NomEntitÃ©Amie' ]
     */
    abstract public function setProcessParameters(): void;

    public function process(mixed $data, Operation $operation, array $uriVariables = [], array $context = []): mixed
    {
        if (!$data instanceof $this->resourceClass) {
            return $this->decorated->process($data, $operation, $uriVariables, $context);
        }

        /** @var User $user */
        $user = $this->security->getUser();

        // âš ï¸ Si l'objet est public (owner = null), on interdit toute modification
        if ($data->getOwner() !== null && $data->getOwner() !== $user) {
            throw new \RuntimeException('Vous ne pouvez modifier que vos propres objets.');
        }

        // âš ï¸ Si l'objet est public et que c'est une modification â†’ interdit
        if ($data->getOwner() === null && $operation->getMethod() !== 'POST') {
            throw new \RuntimeException('Les objets partagÃ©s sont en lecture seule.');
        }

        // ðŸ“ Lors dâ€™un POST â†’ assigner lâ€™utilisateur
        if ($operation->getMethod() === 'POST') {
            $data->setOwner($user);
        }

        // ðŸ”„ Relations imbriquÃ©es (ManyToMany)
        foreach ($this->methods as $method => $relatedEntityLabel) {
            foreach ($data->$method() as $sub) {
                if ($sub->getId()) {
                    if ($sub->getOwner() !== null && $sub->getOwner() !== $user) {
                        throw new \RuntimeException("Un(e) $relatedEntityLabel liÃ©(e) ne vous appartient pas.");
                    }
                } else {
                    $sub->setOwner($user);
                }
            }
        }

        return $this->decorated->process($data, $operation, $uriVariables, $context);
    }
}
